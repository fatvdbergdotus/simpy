    probabilistic
    cost distribution normal(500,50)
    cost range [380,600]
}

component ControllerBoard {
    probabilistic
    cost distribution normal(400,80)
    cost range [200,600]
}

component InkSystem {
    probabilistic
    cost distribution ecdf [200,220,1000]
}

component PaperFeed {
    nondeterministic
    cost range [100,300]
}

component CarriageSystem {
    nondeterministic
    cost range [150,400]
}

component PowerSupply {
    nondeterministic
    cost range [80,200]
}

component Sensors {
    nondeterministic
    cost range [20,100]
}

component Frame {
    nondeterministic
    cost range [200,700]
}

component ControlPanel {
    nondeterministic
    cost range [80,250]
}

component Printer {

    component PrintHead
    component ControllerBoard
    component InkSystem
    component PaperFeed
    component CarriageSystem
    component PowerSupply
    component Sensors
    component Frame
    component ControlPanel

    cost = sum(components.cost)
}
